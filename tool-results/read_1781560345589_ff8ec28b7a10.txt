   251→function ProductCard({ product }: { product: Product }) {
   252→  const { setView, setSelectedProduct, addToCart, addToWishlist, removeFromWishlist, isInWishlist, locale } = useStore()
   253→  const [imgError, setImgError] = useState(false)
   254→  const discount = calculateDiscount(product.price, product.comparePrice)
   255→  const wishlisted = isInWishlist(product.id)
   256→
   257→  const handleQuickAdd = (e: React.MouseEvent) => {
   258→    e.stopPropagation()
   259→    const sizes = product.sizes.split(',')
   260→    const colors = product.colors.split(',')
   261→    addToCart({ productId: product.id, name: product.name, subtitle: product.subtitle, price: product.price, comparePrice: product.comparePrice || undefined, image: product.images, quantity: 1, size: sizes[0] || 'M', color: colors[0] || 'Default' })
   262→    toast.success(`${product.name} added to cart!`)
   263→  }
   264→
   265→  const handleWishlist = (e: React.MouseEvent) => {
   266→    e.stopPropagation()
   267→    if (wishlisted) { removeFromWishlist(product.id); toast.info('Removed from wishlist') }
   268→    else { addToWishlist({ productId: product.id, name: product.name, price: product.price, image: product.images }); toast.success('Added to wishlist!') }
   269→  }
   270→
   271→  return (
   272→    <Card className="group cursor-pointer hover:shadow-xl transition-all duration-300 border-gray-100 overflow-hidden" onClick={() => { setSelectedProduct(product.id); setView('product') }}>
   273→      <div className="relative aspect-[3/4] bg-gray-50 overflow-hidden">
   274→        {imgError ? (
   275→          <div className="w-full h-full flex items-center justify-center bg-gradient-to-br from-rose-100 to-fuchsia-100"><ImageIcon className="w-12 h-12 text-rose-300" /></div>
   276→        ) : (
   277→          <img src={product.images} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500" onError={() => setImgError(true)} loading="lazy" />
   278→        )}
   279→        <div className="absolute top-2 left-2 flex flex-col gap-1">
   280→          {discount > 0 && <Badge className="bg-gradient-to-r from-rose-500 to-fuchsia-600 text-white text-xs">-{discount}%</Badge>}
   281→          {product.isNew && <Badge className="bg-emerald-500 text-white text-xs">New</Badge>}
   282→          {product.featured && <Badge className="bg-amber-500 text-white text-xs">Featured</Badge>}
   283→        </div>
   284→        <Button size="sm" variant="ghost" className={`absolute top-2 right-2 h-8 w-8 p-0 rounded-full ${wishlisted ? 'bg-rose-50 text-rose-600' : 'bg-white/80 text-gray-400 opacity-0 group-hover:opacity-100'} transition-all shadow-sm`}
   285→          onClick={handleWishlist}>
   286→          <Heart className={`w-4 h-4 ${wishlisted ? 'fill-rose-500' : ''}`} />
   287→        </Button>
   288→        <Button size="sm" className="absolute bottom-2 right-2 bg-gradient-to-r from-rose-500 to-fuchsia-600 hover:from-rose-600 hover:to-fuchsia-700 text-white opacity-0 group-hover:opacity-100 transition-all shadow-lg" onClick={handleQuickAdd}>
   289→          <Plus className="w-4 h-4" />
   290→        </Button>
   291→      </div>
   292→      <CardContent className="p-3">
   293→        <div className="flex items-center gap-1.5 mb-1">
   294→          <p className="text-[10px] text-rose-600 font-semibold uppercase tracking-wider">{product.category}</p>
   295→          {product.subtitle && <span className="text-[10px] text-gray-400">· {product.subtitle}</span>}
   296→        </div>
   297→        <h3 className="font-semibold text-sm text-gray-900 line-clamp-1 mb-1">{product.name}</h3>
   298→        <p className="text-xs text-gray-500 line-clamp-2 mb-2">{product.description}</p>
   299→        <div className="flex items-center gap-1 mb-2">
   300→          {Array.from({ length: 5 }).map((_, i) => (
   301→            <Star key={i} className={`w-3 h-3 ${i < Math.floor(product.rating) ? 'fill-amber-400 text-amber-400' : 'text-gray-200'}`} />
   302→          ))}
   303→          <span className="text-xs text-gray-500 ml-1">({product.reviewCount})</span>
   304→        </div>
   305→        <div className="flex items-center gap-2">
   306→          <span className="font-bold text-gray-900">${product.price.toFixed(2)}</span>
   307→          {product.comparePrice && <span className="text-sm text-gray-400 line-through">${product.comparePrice.toFixed(2)}</span>}
   308→        </div>
   309→      </CardContent>
   310→    </Card>
   311→  )
   312→}

   315→function ProductGrid() {
   316→  const [products, setProducts] = useState<Product[]>([])
   317→  const [categories, setCategories] = useState<Category[]>([])
   318→  const [loading, setLoading] = useState(true)
   319→  const { selectedCategory, setSelectedCategory, searchQuery, setSearchQuery, sortBy, setSortBy } = useStore()
   320→
   321→  useEffect(() => {
   322→    async function fetchData() {
   323→      setLoading(true)
   324→      try {
   325→        const params = new URLSearchParams()
   326→        if (selectedCategory !== 'all') params.set('category', selectedCategory)
   327→        if (searchQuery) params.set('search', searchQuery)
   328→        const res = await fetch(`/api/products?${params}`)
   329→        const data = await res.json()
   330→        setProducts(data.products || [])
   331→        if (data.categories) setCategories(data.categories)
   332→      } catch (e) { console.error(e) }
   333→      setLoading(false)
   334→    }
   335→    fetchData()
   336→  }, [selectedCategory, searchQuery])
   337→
   338→  const sortedProducts = [...products].sort((a, b) => {
   339→    switch (sortBy) {
   340→      case 'price_asc': return a.price - b.price
   341→      case 'price_desc': return b.price - a.price
   342→      case 'rating': return b.rating - a.rating
   343→      default: return 0
   344→    }
   345→  })
   346→
   347→  return (
   348→    <div>
   349→      <CategoryBar categories={categories} selected={selectedCategory} onSelect={setSelectedCategory} />
   350→      <div className="max-w-7xl mx-auto px-4 py-6">
   351→        <div className="flex flex-col sm:flex-row gap-3 mb-6">
   352→          <div className="relative flex-1">
   353→            <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-400" />
   354→            <Input placeholder="Search for dresses, tops, shoes..." value={searchQuery} onChange={(e) => setSearchQuery(e.target.value)} className="pl-10 h-11" />
   355→          </div>
   356→          <Select value={sortBy} onValueChange={setSortBy}>
   357→            <SelectTrigger className="w-[180px] h-11"><Filter className="w-4 h-4 mr-2 text-gray-400" /><SelectValue /></SelectTrigger>
   358→            <SelectContent>
   359→              {SORT_OPTIONS.map((opt) => <SelectItem key={opt.value} value={opt.value}>{opt.label}</SelectItem>)}
   360→            </SelectContent>
   361→          </Select>
